  12. //和快排类似, 找一个枢纽元,把小于枢纽元的数放在枢纽元左边, 返回枢纽元的位置
  13. int Partion(int *nArray, int nBegin, int nEnd)
  14. {
  15. 	if(nArray == NULL|| nBegin <0 || nEnd<0)
  16. 		throw exception("ERROR");
  18. 	int nPivot = nArray[nBegin];  //枢纽元
  19. 	int nLeft = nBegin;
  20. 	int nRight = nEnd;
  21. 	while(nLeft != nRight)
  22. 	{
  23. 		while(nArray[nRight]>=nPivot && nLeft < nRight)
  24. 			--nRight;
  25. 		while(nArray[nLeft]<=nPivot && nLeft < nRight)
  26. 			++nLeft;
  27. 		if(nLeft<nRight)
  28. 		{
  29. 			Swap(nArray[nLeft], nArray[nRight]);
  30. 		}
  31. 	}
  32. 	Swap(nArray[nBegin], nArray[nLeft]);
  33. 	return nLeft;
  34. }

  37. {

  44. 	//找出排在中间的那个数的值就是.
  45. 	int nNum = Partion(nArray, nBegin, nEnd);

  48. 		if(nNum > nMiddle)

  50. 			nNum = Partion(nArray, nBegin, nNum-1);
  51. 		}

  53. 		{
  54. 			nNum = Partion(nArray, nNum+1, nEnd);
  55. 		}
  56. 	}
